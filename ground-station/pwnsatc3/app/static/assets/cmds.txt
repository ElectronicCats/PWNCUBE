# GS RADIO CONFIG
COMMAND RDCONNECTOR FREQUENCY BIG_ENDIAN "Set GS Frequency"
  ID_PARAMETER PKTIDD 8 8  UINT  0x00    0x00  0x00 "Set Frequency command"
  APPEND_PARAMETER    VALUE   32    FLOAT  433   960 916      "Frequency in MHz"
  UNITS MHz MHz

# SATELLITE COMMANDS
COMMAND RDCONNECTOR STATUS BIG_ENDIAN "Get Status"
  ID_PARAMETER PKTIDD 8 8  UINT  0x01    0x01  0x01 "Command to Query Status"

COMMAND RDCONNECTOR GET_TEMP BIG_ENDIAN "Get TEMP"
  ID_PARAMETER PKTIDD 8 8  UINT  0x02    0x02  0x02 "Command to Query the Satellite Temperature"
  RESPONSE RDCONNECTOR TEMP

COMMAND RDCONNECTOR GET_GYRO BIG_ENDIAN "Get GYRO"
  ID_PARAMETER PKTIDD 8 8  UINT  0x03    0x03  0x03 "Command to Query the Satellite Gyroscope"
  RESPONSE RDCONNECTOR GYRO

COMMAND RDCONNECTOR GET_TM BIG_ENDIAN "Get TELEMETRY"
  ID_PARAMETER PKTIDD 8 8  UINT  0x04    0x04  0x04 "Command to Query the Satellite Temperature"
  RESPONSE RDCONNECTOR TM

COMMAND RDCONNECTOR SEND_MESSAGE BIG_ENDIAN "Send Message"
  ID_PARAMETER PKTID 8 8  UINT  0x05    0x05  0x05 "Send Broadcast Message from the Satellite"
  APPEND_PARAMETER CMD_STRING 0 STRING "SEND_MESSAGE" "Message to send"
  RESPONSE RDCONNECTOR MESSAGE

COMMAND RDCONNECTOR PING BIG_ENDIAN "Ping Satellite"
  ID_PARAMETER PKTIDD 8 8  UINT  0x09    0x09  0x09 "Ping Command"
  RESPONSE RDCONNECTOR PING
